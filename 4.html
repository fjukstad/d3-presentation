var svg = d3.select("body")
                    .append("svg")
                    .attr("width", "720")
                    .attr("height", "120"); 

    var circle = svg.selectAll("circle")
        .data([20,40,70,90], function(d) { return d; });
     
    circle.enter().append("circle")
        .attr("r", function(d){return d/2;})
        .attr("cy", 60) 
        .attr("cx", function(d,i){return i * 100 + 30;});
     
    circle.exit().remove();
