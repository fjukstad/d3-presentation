var dataset = [{x:10, y:10},
                {x:20, y:80}];

var svg = d3.select("body")
            .append("svg")
            .attr("width", "300")
            .attr("height", "200");

var circle = svg.selectAll("circle")
                .data(dataset)
                
circle.enter().append("circle")
      .attr("r", "4")
      .attr("cx", function(d) {
            return d.x
       })
      .attr("cy", function(d) {
            return d.y
      });
